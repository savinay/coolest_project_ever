define([
	"./Drawable",
	"./Point"
	], function(Drawable, Point) {

Stroke.prototype = new Drawable();
Stroke.prototype.constructor = Stroke;

/**
 * Stroke sketch object, based on the Drawable element
 *
 * @class
 * @memberOf container
 * 
 * @constructor
 * @param {Point[]} [points] List of points belonging to the stroke
 * @param {number} [time] Timestamp of the stroke in milliseconds
 * @param {string} [id] GUID of the stroke
 */
function Stroke(points, time, id) {

	// Call super
	Drawable.call(this, time, id);


	// Functions

	/** @access public */

	/**
	 * Setter for points
	 *
	 * @param {Point[]} newPoints List of points to set to the stroke
	 */
	this.setPoints = function(newPoints) {
		this.points = newPoints;
	}

	/**
	 * Getter for points
	 *
	 * @return {Point[]} List of points belonging to the stroke
	 */
	this.getPoints = function() {
		return this.points;
	}

	/**
	 * Get the number of points in the stroke
	 *
	 * @return {number} Number of points in the stroke
	 */
	this.getNumPoints = function() {
		return this.points.length;
	}

	/**
	 * Gets the point reference at index i in the points list
	 *
	 * @param {number} i Index of point to retrieve from the points list (indexed from 0)
	 *
	 * @return {Point} Point object at index i
	 */
	this.getPointAt = function(i) {
		return this.points[i];
	}

	/**
	 * Add a new point or list of points to the stroke's point list
	 *
	 * @param {Point|Point[]} newPoint Point or list of points to be added to the list of points
	 */
	this.addPoint = function(newPoint) {
		if (newPoint instanceof Array) {
			this.points = this.points.concat(newPoint);
		}
		else {
			this.points.push(newPoint);
		}
	}


	// Members

	/** @access public */

	if (!points) {
		this.points = [];
	}
	else {
		this.points = points;
	}

};

return Stroke;
});
