define([
	"./Drawable"
	], function(Drawable) {

Point.prototype = new Drawable();
Point.prototype.constructor = Point;

/**
 * Smallest representation of a drawable object
 *
 * @class
 * @memberOf container
 * 
 * @constructor
 * @param {number} x Point's x position
 * @param {number} y Point's y position
 * @param {number} [time] Timestamp, given to Drawable, in ms
 * @param {string} [id] GUID of the point
 */
function Point(x, y, time, id) {

	// Call super
	Drawable.call(this, time, id);


	// Functions
	
	/** @access public */

	/**
	 * Setter for x
	 * 
	 * @param {number} newX New x position to replace existing Point x position
	 */
	this.setX = function(newX) {
		this.x = newX;
	}

	/**
	 * Setter for y
	 * 
	 * @param {number} newY New y position to replace existing Point y position
	 */
	this.setY = function(newY) {
		this.y = newY;
	}

	/**
	 * Getter for x
	 * 
	 * @return {number} Point's x position
	 */
	this.getX = function() {
		return this.x;
	}

	/**
	 * Getter for y
	 * 
	 * @return {number} Point's y position
	 */
	this.getY = function() {
		return this.y;
	}

	/** @access public */

	/**
	 * Compute the distance from the current point to an arbitrary x and y location or another Point object
	 *
	 * @param {number|Point} x Desired x-value for distance or the other point
	 * @param {number} y Desired y-value for distance; this is ignored if the first argument was another point
	 *
	 * @return {number} Distance between the current point and the other point or x,y location
	 */
	this.distance = function(x, y) {
		// Check if numbers of a Point have been passed in
		if (x instanceof Point) {
			var yVal = x.y;
			var xVal = x.x;
		}

		var Xdist = this.x - xVal;
		var Ydist = this.y - yVal;

		return Math.sqrt(Xdist * Xdist + Ydist * Ydist);
	}

	/**
	 * Compute the squared distance from the current point to an arbitrary x and y location or another Point object
	 *
	 * @param {number|Point} x Desired x-value for squared distance or the other point
	 * @param {number} y Desired y-value for squared distance; this is ignored if the first argument was another point
	 *
	 * @return {number} Squared distance between the current point and the other point or x,y location
	 */
	this.distanceSquared = function(x, y){
		// Check if numbers of a Point have been passed in
		if (x instanceof Point) {
			var y = x.y;
			var x = x.x;
		}
		return ((this.x-x)*(this.x-x)+(this.y-y)*(this.y-y));
	}


	// Members

	/** @access public */

	this.x = x;
	this.y = y;

};

return Point;
});
