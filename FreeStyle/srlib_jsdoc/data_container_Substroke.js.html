define([
	"./Drawable",
	"./Point",
	"./Stroke"
	], function(Drawable, Point, Stroke) {

Substroke.prototype = new Drawable();
Substroke.prototype.constructor = Substroke;

/**
 * Derivative of the Stroke object, but actually part of Drawable.
 *
 * @class
 * @memberOf container
 * 
 * @constructor
 * @param {Stroke} parent Parent stroke to which substroke object belongs
 * @param {Point[]} [points] Point array of the points in the substroke
 * @param {number} [time] Javascript Timestamp, ms
 * @param {string} [id] GUID of substroke
 */
function Substroke(parent, points, time, id) {

	// Call super
	Drawable.call(this, time, id);


	// Functions

	/** @access public */

	/**
	 * Setter for parent
	 * 
	 * @param {Stroke} newParent New Parent stroke object
	 */
	this.setParent = function(newParent) {
		this.parent = newParent;
	}

	/**
	 * Setter for points
	 * 
	 * @param {Point[]} newPoints Points to add to the substroke
	 */
	this.setPoints = function(newPoints) {
		this.points = newPoints;
	}

	/**
	 * Getter for parent
	 * 
	 * @return {Stroke} Returns parent of the substroke
	 */
	this.getParent = function() {
		return this.parent;
	}

	/**
	 * Getter for points
	 * 
	 * @return {Point[]} Substroke point list
	 */
	this.getPoints = function() {
		return this.points;
	}

	/**
	 * Get the number of points in the stroke
	 *
	 * @return {number} Number of points in the stroke
	 */
	this.getNumPoints = function() {
		return this.points.length;
	}

	/**
	 * Add a point or list of points to the point list of the substroke
	 * 
	 * @param {Point|Point[]} newPoint New point or list of points to add to the substroke
	 */
	this.addPoint = function(newPoint) {
		if (newPoint instanceof Array) {
			this.points = this.points.concat(newPoint);
		}
		else {
			this.points.push(newPoint);
		}
	}


	// Members

	/** @access public */

	this.parent = parent;

	if(!points){
		this.points = [];
	}
	else{
		this.points = points;
	}

};

return Substroke;
});
