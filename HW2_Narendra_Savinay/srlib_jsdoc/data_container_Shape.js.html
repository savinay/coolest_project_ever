define([
	"./Drawable",
	"./Point",
	"./Stroke",
	"./Substroke"
	], function(Drawable, Point, Stroke, Substroke) {

Shape.prototype = new Drawable();
Shape.prototype.constructor = Shape;

/**
 * Shape object contains other drawable elements and attaches an interpretation
 *
 * @class
 * @memberOf container
 * 
 * @constructor
 * @param {Drawable[]} [subElements] List of drawable subelements
 * @param {string} [interpretation] Classification of the shape (e.g. triangle, circle, truss, etc.)
 * @param {number} [confidence] Statistical confidence (0.0 <= confidence <= 1.0)
 * @param {number} [time] JS timestamp, ms
 * @param {string} [id] GUID of the shape
 */
function Shape(subElements, interpretation, confidence, time, id) {

	// Call super
	Drawable.call(this, time, id);


	// Functions

	/** @access public */

	/**
	 * Setter for subElements
	 * 
	 * @param {Drawable[]} newSubElements New subElements array to set inside the shape
	 */
	this.setSubElements = function(newSubElements) {
		this.subElements = newSubElements;
	}

	/**
	 * Setter for interpretation
	 * 
	 * @param {string} newInterpretation New interpretation of the shape
	 */
	this.setInterpretation = function(newInterpretation) {
		this.interpretation = newInterpretation;
	}

	/**
	 * Setter for confidence
	 * 
	 * @param {number} newConfidence New Confidence of the interpretation
	 */
	this.setConfidence = function(newConfidence) {
		this.confidence = newConfidence;
	}

	/**
	 * Getter for subElements list
	 * 
	 * @return {Drawable[]} List of subelements referenced by the shape
	 */
	this.getSubElements = function() {
		return this.subElements;
	}

	/**
	 * Getter for interpretation
	 * 
	 * @return {string} Interpretation of the shape
	 */
	this.getInterpretation = function() {
		return this.interpretation;
	}

	/**
	 * Getter for confidence
	 * 
	 * @return {number} Confidence of the interpretation
	 */
	this.getConfidence = function() {
		return this.confidence;
	}

	/**
	 * Appends a subElement or list of subElements to end of subElements array
	 * 
	 * @param {Drawable|Drawable[]} newSubElement New drawable object or list of drawable objects to add to the subElements list
	 */
	this.addSubElement = function(newSubElement) {
		if (newSubElement instanceof Array) {
			this.points = this.points.concat(newSubElement);
		}
		else {
			this.points.push(newSubElement);
		}
	}


	// Members

	/** @access public */

	if (!subElements) {
		this.subElements = [];
	}
	else {
		this.subElements = subElements;
	}

	if (!interpretation) {
		this.interpretation = "";
	}
	else {
		this.interpretation = interpretation;
	}

	if (!confidence) {
		this.confidence = 0;
	}
	else {
		this.confidence = confidence;
	}

};

return Shape;
});
