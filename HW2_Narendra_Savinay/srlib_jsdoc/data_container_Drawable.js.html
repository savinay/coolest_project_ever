define([], function() {

/**
 * Base object for any drawable element, handles GUIDs, timestamps, bounding box, and other attributes
 *
 * @class
 * @memberOf container
 * 
 * @constructor
 * @param {number} [time] Timestamp of the drawable element to be created in milliseconds
 * @param {string} [id] GUID of the drawable element to be created
 */
function Drawable(time, id) {

	// Functions
	/** @access private */

	/**
	 * Generates an RFC4122v4-compliant GUID
	 *
	 * @return {string} GUID
	 */
	var setId = function() {
		return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function(c) {
			var r = Math.random()*16|0, v = c === 'x' ? r : (r&0x3|0x8);
			return v.toString(16);
		});
	}

	/** @access public */

	/**
	 * Getter for id
	 *
	 * @return {string} GUID of drawable element
	 */
	this.getId = function() {
		return id;
	}

	/**
	 * Generates a millisecond-granularity timestamp from the current time
	 * 
	 * @param {number} time Timestamp to set as the time for the object
	 */
	this.setTime = function(time) {
		this.time = time;
	}

	/**
	 * Getter for time
	 *
	 * @return {number} Timestamp of drawable element
	 */
	this.getTime = function() {
		return this.time;
	}


	// Members

	/** @access private */

	if (!id) {
		var id = setId();
	}
	else {
		var id = id;
	}

	/** @access public */

	if (!time) {
		this.setTime(new Date().getTime());
	}
	else {
		this.time = time;
	}

};

return Drawable;
});
